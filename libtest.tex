\newcommand{\testprf}[1]{
    \begin{#1}
        Let $V\subS U$.
    
        \FirstWay Let $V$ be open in $X$. Then, $U\cap V = V$ is open, and so
        $V$ is open in $U$.
        
        \OtherWay Let $V$ be open in $U$. Then, $V = U\cap W$ for some open
        $W$ in $X$. Because $U$ is open in $X$, $U\cap W$ is open. Hence $V$ is
        open in $X$.
    \end{#1}
}
