% % %
% Theorem Enviroments
%
% each environment comes in one of three forms:
%   Sthm: Section numbered
%   thm: numbered
%   thm*: unnumbered
% % %

%% Imports %%
\usepackage{amsthm}
\usepackage{wasysym}

% shorthand for theorem generation
\newcommand{\mkthm}[2]{%
    \newtheorem{S#1}{#2}[section]
    \newtheorem{#1}{#2}
    \newtheorem*{#1*}{#2}
}

\mkthm{thm}{Theorem}
\mkthm{lem}{Lemma}
\mkthm{prop}{Proposition}
\mkthm{cor}{Corollary}

\theoremstyle{definition}

\mkthm{defn}{Definition}
\mkthm{axi}{Axiom}
\mkthm{conj}{Conjecture}
\mkthm{example}{Example}
\mkthm{exercise}{Exercise}
\mkthm{fact}{Fact}

\theoremstyle{remark}

\mkthm{rem}{Remark}
\mkthm{note}{Note}

\newenvironment{solution}
  {\begin{proof}[Solution]}
  {\end{proof}}
  
\newcommand{\useHappyQED}{\renewcommand{\qedsymbol}{\smiley{}}}
